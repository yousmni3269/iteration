``` r
mean_and_sd = function(x) {
  
  mean_x = mean(x)
  sd_x = sd(x)
  
  out_df = 
    tibble(
      mean = mean_x,
      sd = sd_x 
    )
  
  return(out_df)
  
}
```

nsduh_table_format = function(html, table_num){
  
  out_table = 
    html |>
    html_table() |>
    nth(table_num) |>
    slice(-1) |>
    select(-contains("P Value"))
  
  return(out_table)
  
}
